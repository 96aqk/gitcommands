//initialize local git repo
git init 

//adds files to index
git add <file> 

//checks what files are in staging area
git status

//commits changes in index
git commit -m "your_comments"

//push to remote repo
git push

//pull latest from remote directory
git pull

//clone repo into a new directory
git clone

git checkout -b branch_name

//checks current branch
git branch -v 

//deletes branch
git branch -D branch_name

//checks history
git log
